## Key Decisions Made

/*
 * AudioHelper.cs
 * Author: ST10467645
 * Date: May 2026
 * Module: PROG6221 - Programming 2A
 * Part: 2 - GUI Chatbot
 *
 * Purpose:
 * This class is responsible for playing the voice greeting WAV file
 * when the application first opens. It is kept in its own separate class
 * so that the audio logic does not clutter the main form code, which
 * follows the principle of separation of concerns in OOP.
 *
 * Why Play() instead of PlaySync():
 * PlaySync() would block the main thread and freeze the entire window
 * until the audio finishes playing, which would make the app feel broken.
 * Play() runs the audio in the background so the form loads at the same time.
 *
 * The method returns a status string so the caller (Form1) can display
 * either a success message or an error message in the audio label on the splash screen.
 *
 * References:
 * Microsoft (2024). SoundPlayer Class.
 * Available at: https://learn.microsoft.com/en-us/dotnet/api/system.media.soundplayer
 * Web Dev Tutor (2024). C# System.Media.SoundPlayer.
 * Available at: https://www.webdevtutor.net/blog/c-sharp-systemmediasoundplayer
 */

using System;
using System.Media;

namespace ST10467645_PROG6221_Part2
{
    public class AudioHelper
    {
        public static string PlayGreeting()
        {
            /*
             * PlayGreeting method:
             *
             * - Attempts to load and play the greeting.wav audio file.
             * - The WAV file must be in the same output folder as the .exe.
             *   This is ensured by setting "Copy to Output Directory = Copy always"
             *   on the file in Solution Explorer.
             * - Uses Play() so the audio runs in the background without freezing the form.
             * - Returns a status string that Form1 uses to update the audio label:
             *     Success: shows "Playing voice greeting... (greeting.wav)"
             *     Failure: shows a warning that the file could not be found.
             * - The try-catch means a missing file never crashes the application.
             */

            try
            {
                // Creating a SoundPlayer and pointing it at the WAV file in the output folder.
                SoundPlayer player = new SoundPlayer("greeting.wav");

                // Playing in the background so the form continues loading simultaneously.
                player.Play();

                // Returning a success message for the audio label on the splash screen.
                return "  ●   Playing voice greeting...  (greeting.wav)";
            }
            catch (Exception ex)
            {
                // Logging the actual technical error to the Output window for debugging purposes.
                Console.WriteLine($"Audio greeting could not be played: {ex.Message}");

                // Returning a user-friendly error message so the label updates instead of staying blank.
                return "  ⚠   Audio file not found - greeting.wav missing from output folder";
            }
        }
    }
}

using System;
using System.Collections.Generic;

namespace ST10467645_PROG6221_Part2
{
    public class ChatEngine
    {
        // Public properties — Form1 reads these to update the status bar.
        // Private set means only ChatEngine can change them.
        public string UserName         { get; private set; }
        public string CurrentTopic     { get; private set; } = "General";
        public string CurrentSentiment { get; private set; } = "Neutral";

        // Memory dictionary — stores key-value pairs for the session.
        // e.g. memory["name"] = "Kandyce", memory["favouriteTopic"] = "privacy"
        private Dictionary<string, string> memory;

        // Tracks last topic discussed — used by conversation flow.
        private string lastTopic = "";

        // Single shared Random instance for better randomness.
        private Random random;

        // Five tip lists, one per topic, 7 tips each.
        private List<string> passwordTips;
        private List<string> phishingTips;
        private List<string> browsingTips;
        private List<string> scamTips;
        private List<string> privacyTips;

        // ═══════════════════════════════════════════════════════
        //  CONSTRUCTOR
        // ═══════════════════════════════════════════════════════

        public ChatEngine(string name)
        {
            /*
             * Called once when user enters valid name.
             * Sets up memory, Random, and all tip lists.
             */

            UserName       = name;
            random         = new Random();
            memory         = new Dictionary<string, string>();
            memory["name"] = name;

            LoadPasswordTips();
            LoadPhishingTips();
            LoadBrowsingTips();
            LoadScamTips();
            LoadPrivacyTips();
        }

        // ═══════════════════════════════════════════════════════
        //  MAIN RESPONSE METHOD — 10-step priority chain
        // ═══════════════════════════════════════════════════════

        public string GetResponse(string userInput)
        {
            /*
             * Single public entry point. Form1 calls this every message.
             * Processes input through 10-step priority chain.
             * Returns immediately when a match is found.
             * ToLower() once at start so all Contains() checks are case-insensitive.
             *
             * Step 1  - Reject blank/short input
             * Step 2  - Memory recall ("what am I interested in")
             * Step 3  - Small talk (greetings, farewells, thanks)
             * Step 4  - Satisfaction responses (yes/no)
             * Step 5  - Conversation flow (tell me more)
             * Step 6  - Context reply after emotion follow-up
             * Step 7  - Sentiment detection
             * Step 8  - Topic keywords
             * Step 9  - Memory storage (I am interested in...)
             * Step 10 - Default fallback
             */

            string input = userInput.ToLower().Trim();

            // Step 1
            if (string.IsNullOrWhiteSpace(input) || input.Length < 2)
                return "I did not quite catch that! Could you type a bit more so I can help you?";

            // Step 2 — memory recall first to prevent it being caught by awaitingContext
            if (input.Contains("what am i interested in") || input.Contains("what did i say") ||
                input.Contains("do you remember") || input.Contains("what do you know about me"))
            {
                CurrentTopic = "Memory";
                if (memory.ContainsKey("favouriteTopic"))
                    return $"I remember that you told me you are interested in {memory["favouriteTopic"]}, {UserName}!\r\n" +
                           "Would you like me to share a tip on that topic?";
                else
                    return $"I know your name is {UserName}!\r\n" +
                           "You have not told me your favourite topic yet.\r\n" +
                           "Say something like 'I am interested in privacy' and I will remember it.";
            }

            // Step 3
            string smallTalk = CheckSmallTalk(input);
            if (smallTalk != null) return smallTalk;

            // Step 4
            if (input == "yes" || input.Contains("yes that") || input.Contains("satisfied") ||
                input.Contains("that helped") || input.Contains("perfect"))
            {
                CurrentTopic = lastTopic; CurrentSentiment = "Positive";
                return $"Great! I am glad that helped, {UserName}.\r\n" +
                       "Feel free to ask me anything else or click a topic button above!";
            }

            if (input == "no" || input.Contains("not helpful") || input.Contains("not satisfied"))
            {
                CurrentTopic = lastTopic; CurrentSentiment = "Neutral";
                return $"I am sorry about that, {UserName}!\r\n" +
                       "Could you tell me a bit more about what you need?";
            }

            // Step 5
            string followUp = CheckConversationFlow(input);
            if (followUp != null) return followUp;

            // Step 6 — context reply after emotion follow-up question
            if (memory.ContainsKey("awaitingContext") && memory["awaitingContext"] == "true")
            {
                memory["awaitingContext"] = "false";
                string stored = memory.ContainsKey("pendingSentiment") ? memory["pendingSentiment"] : "Neutral";
                CurrentSentiment = stored;
                return GetSentimentWithContext(stored, input);
            }

            // Step 7
            string sentiment = DetectSentiment(input);
            if (sentiment != "Neutral")
            {
                CurrentSentiment = sentiment;
                if (ContainsCyberTopic(input))
                    return GetSentimentWithContext(sentiment, input);

                memory["pendingSentiment"] = sentiment;
                memory["awaitingContext"]  = "true";
                CurrentTopic = "Sentiment";
                return GetEmotionFollowUpQuestion(sentiment);
            }

            // Step 8
            string topicResponse = CheckTopicKeywords(input);
            if (topicResponse != null) return topicResponse;

            // Step 9
            string memoryResponse = CheckMemoryInput(input);
            if (memoryResponse != null) return memoryResponse;

            // Step 10
            CurrentTopic = "Unknown"; CurrentSentiment = "Neutral";
            return $"I am not sure I understood that, {UserName}. Could you try rephrasing?\r\n" +
                   "You can ask about passwords, phishing, safe browsing, scams, or privacy,\r\n" +
                   "or just tell me how you are feeling!";
        }

        // ═══════════════════════════════════════════════════════
        //  HELPER — checks if input contains any cyber topic keyword
        // ═══════════════════════════════════════════════════════

        private bool ContainsCyberTopic(string input)
        {
            return input.Contains("password") || input.Contains("phish") ||
                   input.Contains("scam")     || input.Contains("privacy") ||
                   input.Contains("brows")    || input.Contains("hack") ||
                   input.Contains("virus")    || input.Contains("malware") ||
                   input.Contains("security") || input.Contains("data") ||
                   input.Contains("email")    || input.Contains("account") ||
                   input.Contains("fraud")    || input.Contains("vpn");
        }

        // ═══════════════════════════════════════════════════════
        //  EMOTION FOLLOW-UP QUESTION
        // ═══════════════════════════════════════════════════════

        private string GetEmotionFollowUpQuestion(string sentiment)
        {
            // Acknowledges emotion, asks what caused it.
            // User's next message is caught by awaitingContext check.
            switch (sentiment)
            {
                case "Worried":
                    return $"I hear you, {UserName} - feeling worried is completely valid.\r\n" +
                           "What is making you worried? Passwords, phishing, scams, or something else?";
                case "Frustrated":
                    return $"I totally get that, {UserName}!\r\n" +
                           "What is causing the frustration? Tell me more and I will try to help.";
                case "Excited":
                    return $"That is great to hear, {UserName}!\r\n" +
                           "What are you excited about? Which topic would you like to explore?";
                case "Curious":
                    return $"I love that curiosity, {UserName}!\r\n" +
                           "What would you like to know more about?";
                case "Confused":
                    return $"No problem at all, {UserName} - cybersecurity can be confusing!\r\n" +
                           "What is confusing you? I will break it down simply.";
                case "Stressed":
                    return $"I am sorry you are feeling stressed, {UserName}.\r\n" +
                           "What is causing it? Let me know and I will help.";
                case "Scared":
                    return $"It makes sense to feel scared, {UserName}.\r\n" +
                           "What has got you feeling that way? Tell me and we will work through it.";
                default:
                    return $"Thank you for sharing that, {UserName}.\r\n" +
                           "Could you tell me more about what is on your mind?";
            }
        }

        // ═══════════════════════════════════════════════════════
        //  SENTIMENT WITH CONTEXT — empathy + relevant tip
        // ═══════════════════════════════════════════════════════

        private string GetSentimentWithContext(string sentiment, string input)
        {
            // Builds empathy opening then appends relevant topic tip.
            string empathy = BuildEmpathyMessage(sentiment);

            if (input.Contains("password") || input.Contains("login"))
            { lastTopic = "Password"; CurrentTopic = "Password"; return empathy + GetPasswordResponse(input); }
            if (input.Contains("phish") || input.Contains("fake email"))
            { lastTopic = "Phishing"; CurrentTopic = "Phishing"; return empathy + GetPhishingResponse(input); }
            if (input.Contains("scam") || input.Contains("fraud"))
            { lastTopic = "Scams"; CurrentTopic = "Scams"; return empathy + GetScamResponse(input); }
            if (input.Contains("privacy") || input.Contains("data") || input.Contains("2fa"))
            { lastTopic = "Privacy"; CurrentTopic = "Privacy"; return empathy + GetPrivacyResponse(input); }
            if (input.Contains("brows") || input.Contains("virus") || input.Contains("vpn") || input.Contains("hack"))
            { lastTopic = "Browsing"; CurrentTopic = "Browsing"; return empathy + GetBrowsingResponse(input); }

            return empathy + $"Click a topic button or ask me about passwords, phishing,\r\n" +
                   "safe browsing, scams, or privacy and I will give you targeted advice!";
        }

        private string BuildEmpathyMessage(string sentiment)
        {
            // Short empathy opener based on emotion. Precedes the relevant tip.
            switch (sentiment)
            {
                case "Worried":   return $"I completely understand, {UserName} - it is smart to be cautious.\r\nHere is something that should help:\r\n\r\n";
                case "Stressed":  return $"I am sorry you are going through that, {UserName}.\r\nHere is something practical that can help:\r\n\r\n";
                case "Scared":    return $"That is a completely natural reaction, {UserName}.\r\nHere is what you need to know:\r\n\r\n";
                case "Frustrated":return $"I totally understand that frustration, {UserName}.\r\nHere is something useful:\r\n\r\n";
                case "Excited":   return $"That enthusiasm is brilliant, {UserName}!\r\nHere is something great to know:\r\n\r\n";
                case "Curious":   return $"Great curiosity, {UserName}!\r\nHere is what you should know:\r\n\r\n";
                case "Confused":  return $"No worries at all, {UserName} - let me break this down:\r\n\r\n";
                default:          return $"Here is something relevant for you, {UserName}:\r\n\r\n";
            }
        }

        // ═══════════════════════════════════════════════════════
        //  SMALL TALK
        // ═══════════════════════════════════════════════════════

        private string CheckSmallTalk(string input)
        {
            // Returns null if nothing matches — chain continues.
            if (input.Contains("how are you") || input.Contains("how r you"))
            { CurrentTopic = "General"; CurrentSentiment = "Positive";
              return $"I am doing great, {UserName}! Always ready to help you stay safe online."; }

            if (input.Contains("your purpose") || input.Contains("what do you do") || input.Contains("who are you"))
            { CurrentTopic = "General"; CurrentSentiment = "Neutral";
              return $"My purpose, {UserName}, is to help you understand cybersecurity and stay safe online.\r\n" +
                     "I cover: Password safety, Phishing, Safe browsing, Scams, and Privacy."; }

            if (input.Contains("what can i ask") || input.Contains("what topics") || input.Contains("help me"))
            { CurrentTopic = "General"; CurrentSentiment = "Neutral";
              return $"You can ask me about:\r\n" +
                     "  - Password safety\r\n  - Phishing\r\n  - Safe browsing\r\n  - Scams\r\n  - Privacy\r\n\r\n" +
                     "Or just tell me how you are feeling and I will respond!"; }

            if (input.Contains("hello") || input.Contains("hi ") || input == "hi" || input.Contains("hey"))
            { CurrentTopic = "General"; CurrentSentiment = "Positive";
              return $"Hello, {UserName}! How can I help you stay safe online today?"; }

            if (input.Contains("goodbye") || input.Contains("bye") || input.Contains("farewell") || input.Contains("see you"))
            { CurrentTopic = "General"; CurrentSentiment = "Positive";
              return $"Goodbye, {UserName}! Stay safe online.\r\n" +
                     "Remember: think before you click and use strong unique passwords!"; }

            if (input.Contains("thank") || input.Contains("thanks") || input.Contains("appreciate"))
            { CurrentTopic = "General"; CurrentSentiment = "Positive";
              return $"You are very welcome, {UserName}! Is there anything else I can help with?"; }

            return null;
        }

        // ═══════════════════════════════════════════════════════
        //  CONVERSATION FLOW
        // ═══════════════════════════════════════════════════════

        private string CheckConversationFlow(string input)
        {
            // Detects follow-up phrases, uses lastTopic to continue.
            bool isFollowUp = input.Contains("tell me more") || input.Contains("another tip") ||
                              input.Contains("more info")    || input.Contains("explain more") ||
                              input.Contains("elaborate")    || input.Contains("continue") ||
                              input.Contains("go on")        || input.Contains("give me a tip") ||
                              input.Contains("give me more") || input.Contains("more please");

            if (!isFollowUp) return null;

            CurrentSentiment = "Curious";

            switch (lastTopic)
            {
                case "Password": CurrentTopic = "Password";
                    return $"Here is another password safety tip, {UserName}:\r\n\r\n" + GetRandomTip(passwordTips);
                case "Phishing": CurrentTopic = "Phishing";
                    return $"Here is another phishing tip, {UserName}:\r\n\r\n" + GetRandomTip(phishingTips);
                case "Browsing": CurrentTopic = "Browsing";
                    return $"Here is another safe browsing tip, {UserName}:\r\n\r\n" + GetRandomTip(browsingTips);
                case "Scams": CurrentTopic = "Scams";
                    return $"Here is another scam awareness tip, {UserName}:\r\n\r\n" + GetRandomTip(scamTips);
                case "Privacy": CurrentTopic = "Privacy";
                    return $"Here is another privacy tip, {UserName}:\r\n\r\n" + GetRandomTip(privacyTips);
                default:
                    return $"Which topic would you like more on?\r\n" +
                           "Password safety, phishing, safe browsing, scams, or privacy?";
            }
        }

        // ═══════════════════════════════════════════════════════
        //  SENTIMENT DETECTION
        // ═══════════════════════════════════════════════════════

        private string DetectSentiment(string input)
        {
            // Uses specific phrases (not single words) to avoid false positives.
            if (input.Contains("i am worried") || input.Contains("i'm worried") ||
                input.Contains("anxious") || input.Contains("i am nervous") || input.Contains("i am scared") ||
                input.Contains("i'm scared") || input.Contains("frightened") || input.Contains("terrified"))
                return "Worried";

            if (input.Contains("i am stressed") || input.Contains("i'm stressed") ||
                input.Contains("overwhelmed") || input.Contains("under pressure") || input.Contains("burned out"))
                return "Stressed";

            if (input.Contains("i am frustrated") || input.Contains("i'm frustrated") ||
                input.Contains("i am angry") || input.Contains("i'm angry") ||
                input.Contains("i am annoyed") || input.Contains("fed up"))
                return "Frustrated";

            if (input.Contains("i am excited") || input.Contains("i'm excited") ||
                input.Contains("i am happy") || input.Contains("i'm happy") || input.Contains("enthusiastic"))
                return "Excited";

            if (input.Contains("i am curious") || input.Contains("i'm curious") ||
                input.Contains("i am interested") || input.Contains("i want to know") || input.Contains("fascinated"))
                return "Curious";

            if (input.Contains("i am confused") || input.Contains("i'm confused") ||
                input.Contains("i don't understand") || input.Contains("i am lost") || input.Contains("don't get it"))
                return "Confused";

            return "Neutral";
        }

        //  TOPIC KEYWORD RECOGNITION
        // ═══════════════════════════════════════════════════════

        private string CheckTopicKeywords(string input)
        {
            // .Contains() checks anywhere in sentence — natural phrasing works.
            if (input.Contains("password") || input.Contains("passcode") || input.Contains("login"))
            { lastTopic = "Password"; CurrentTopic = "Password"; CurrentSentiment = "Neutral";
              return GetPasswordResponse(input); }

            if (input.Contains("phish") || input.Contains("fake email") || input.Contains("spear"))
            { lastTopic = "Phishing"; CurrentTopic = "Phishing"; CurrentSentiment = "Neutral";
              return GetPhishingResponse(input); }

            if (input.Contains("brows") || input.Contains("vpn") || input.Contains("https") ||
                input.Contains("malware") || input.Contains("virus") || input.Contains("hack"))
            { lastTopic = "Browsing"; CurrentTopic = "Browsing"; CurrentSentiment = "Neutral";
              return GetBrowsingResponse(input); }

            if (input.Contains("scam") || input.Contains("fraud") || input.Contains("trick"))
            { lastTopic = "Scams"; CurrentTopic = "Scams"; CurrentSentiment = "Neutral";
              return GetScamResponse(input); }

            if (input.Contains("privacy") || input.Contains("personal info") || input.Contains("2fa") ||
                input.Contains("two factor") || input.Contains("data breach") || input.Contains("identity"))
            { lastTopic = "Privacy"; CurrentTopic = "Privacy"; CurrentSentiment = "Neutral";
              return GetPrivacyResponse(input); }

            return null;
        }

                input.Contains("i am interested") || input.Contains("i'm interested") ||
                input.Contains("my interest is")  || input.Contains("i like") ||
                input.Contains("i love")          || input.Contains("favourite topic") ||
                input.Contains("i want to learn") || input.Contains("i enjoy") ||
                input.Contains("i prefer")        || input.Contains("i care about") ||
                input.Contains("i am into")       || input.Contains("i'm into");

            if (expressingInterest)
            {
                string topic = "";
                if (input.Contains("password"))      topic = "password safety";
                else if (input.Contains("phish"))    topic = "phishing";
                else if (input.Contains("brows"))    topic = "safe browsing";
                else if (input.Contains("scam"))     topic = "scams";
                else if (input.Contains("privacy"))  topic = "privacy";
                else if (input.Contains("security")) topic = "cybersecurity in general";

                if (topic != "")
                {
                    memory["favouriteTopic"] = topic;
                    CurrentTopic = "Memory"; CurrentSentiment = "Positive";
                    return $"Great choice, {UserName}! I will remember you are interested in {topic}.\r\n\r\n" +
                           $"I will connect tips back to {topic} throughout our conversation!";
                }
            }

            if (memory.ContainsKey("favouriteTopic") && input.Contains("tip"))
            {
                string fav = memory["favouriteTopic"];
                CurrentTopic = "Memory"; CurrentSentiment = "Neutral";
                return $"As someone interested in {fav}, {UserName}, here is something relevant:\r\n\r\n" +
                       GetTipForTopic(fav);
            }

            return null;
        }

        private string GetTipForTopic(string topic)
        {
            switch (topic)
            {
                case "password safety":          return GetRandomTip(passwordTips);
                case "phishing":                 return GetRandomTip(phishingTips);
                case "safe browsing":            return GetRandomTip(browsingTips);
                case "scams":                    return GetRandomTip(scamTips);
                case "privacy":                  return GetRandomTip(privacyTips);
                case "cybersecurity in general": return GetRandomTip(phishingTips);
                default:                         return GetRandomTip(phishingTips);
            }
        }

        // ═══════════════════════════════════════════════════════
        //  RANDOM TIP PICKER
        // ═══════════════════════════════════════════════════════

        private string GetRandomTip(List<string> tips)
        {
            // Random.Next(count) generates index 0 to count-1.
            // Creates variety — same question gets different answer each time.
            if (tips == null || tips.Count == 0)
                return "Stay safe online - always think before you click!";

            return tips[random.Next(tips.Count)];
        }
